Architecturer une app — avant d'ecrire une ligne de code.

Avant de coder, on planifie. L'architecture de la mini-app se divise en trois parties : les donnees (state), l'affichage (render), et les actions (handlers). Cette separation rend le code maintenable et previsible.

JavaScript — Architecture de l'app
// ===== STATE =====
// Toutes les donnees en un seul endroit
let portfolio = charger('portfolio', []);
// { ticker, entree, quantite, prixActuel }

// ===== PERSISTENCE =====
function charger(cle, defaut) {
  try { return JSON.parse(localStorage.getItem(cle)) || defaut; }
  catch { return defaut; }
}
function sauvegarder() {
  localStorage.setItem('portfolio', JSON.stringify(portfolio));
}

// ===== API =====
async function actualiserPrix() {
  const ids = portfolio.map(p => p.coingeckoId).join(',');
  const res = await fetch(`https://api.coingecko.com/api/v3/simple/price?ids=${ids}&vs_currencies=usd`);
  const data = await res.json();
  portfolio.forEach(p => {
    p.prixActuel = data[p.coingeckoId]?.usd || p.prixActuel;
  });
}

// ===== RENDER =====
function render() {
  const container = document.querySelector('#app');
  const total = portfolio.reduce((acc, p) => acc + p.prixActuel * p.quantite, 0);
  container.innerHTML = portfolio.map((p, i) => renderItem(p, i)).join('');
}

// ===== HANDLERS =====
async function ajouterPosition() {
  portfolio.push({ ticker: 'BTC', entree: 60000, quantite: 0.5 });
  sauvegarder();
  await actualiserPrix();
  render();
}
State — Source of truth uniqueToutes les donnees dans un seul objet ou tableau (le "state"). Le DOM n'est qu'un reflet de ce state. Quand quelque chose change, tu modifies le state puis tu appelles render() pour mettre a jour le DOM. C'est la base de tous les frameworks modernes (React, Vue, Svelte). Comprendre ca maintenant, c'est comprendre React demain.

JavaScript — Code complet du tracker
// ---- STATE ----
const CLE = 'kora_portfolio';
const COINGECKO_IDS = {
  BTC: 'bitcoin', ETH: 'ethereum', SOL: 'solana',
  BNB: 'binancecoin', XRP: 'ripple'
};
let portfolio = charger();

// ---- PERSISTENCE ----
function charger() {
  try { return JSON.parse(localStorage.getItem(CLE)) || []; }
  catch { return []; }
}
function sauvegarder() {
  localStorage.setItem(CLE, JSON.stringify(portfolio));
}

// ---- API ----
async function refreshPrix() {
  if (!portfolio.length) return;
  const ids = [...new Set(portfolio.map(p => COINGECKO_IDS[p.ticker]).filter(Boolean))].join(',');
  const res = await fetch(`https://api.coingecko.com/api/v3/simple/price?ids=${ids}&vs_currencies=usd`);
  const data = await res.json();
  portfolio.forEach(p => {
    const id = COINGECKO_IDS[p.ticker];
    if (id && data[id]) p.prixActuel = data[id].usd;
  });
  sauvegarder();
}

// ---- RENDER ----
function render() {
  const total = portfolio.reduce((a,p) => a + (p.prixActuel||p.entree)*p.qty, 0);
  document.querySelector('#total').textContent = `${total.toLocaleString('fr-FR', {maximumFractionDigits:0})} $`;
  document.querySelector('#liste').innerHTML =
    portfolio.length ?
    portfolio.map((p, i) => {
      const val = (p.prixActuel||p.entree) * p.qty;
      const pnl = (p.prixActuel ? (p.prixActuel-p.entree)*p.qty : 0);
      return `<div class="item" data-i="${i}">${p.ticker} — ${val.toFixed(0)} $ (${pnl>=0?'+':''}${pnl.toFixed(0)})</div>`;
    }).join('') :
    '<div>Ajoute une position...</div>';
}
La sequence fondamentale1. Modifier le state (portfolio). 2. Sauvegarder (localStorage). 3. Actualiser les prix si necessaire (API). 4. Re-render le DOM. Cet ordre est toujours le meme. C'est la sequence que React, Vue et tous les frameworks moderne automatisent avec leur systeme de reactivite.
